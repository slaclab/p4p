SharedPV methods available (from pvxs/sharedpv.h via installed pvxslibs):
- `pv.isOpen()` → bool
- `pv.fetch()` → Value (current value / data type prototype)
- `pv.onFirstConnect(callback)` / `pv.onLastDisconnect(callback)` — NOT available from custom Source
- ChannelControl methods: `ctrl->onOp(cb)`, `ctrl->onSubscribe(cb)`, `ctrl->onRPC(cb)`
- ConnectOp methods: `cop->connect(Value)`, `cop->onGet(cb)`, `cop->onPut(cb)`
- ExecOp methods: `op->reply(Value)`, `op->reply()`, `op->error(string)`

CRITICAL: `ChannelControl::onOp()` stores a single callback — calling it again replaces. `SharedPV::attach()` calls `op->onOp(pvxs_internal_handler)` internally, which REPLACES any pre-registered onOp. Therefore: do NOT call `pv.attach()` when using GetInterceptSource. Manage PUT/RPC/Subscribe directly in the Source.

For GET fallback (CPP-03, no onGet in handler): call `gop->reply(pv.fetch())` to replicate PVXS default GET behavior.

**Step B: Add to src/pvxs_sharedpv.cpp** — insert the `GetInterceptSource` struct and `attachGetHandler` function after the existing `detachCleanup` function (after line 113, before the closing `}` of namespace p4p).

The struct manages all channel operations without calling `pv.attach()`:

```cpp
struct GetInterceptSource : public server::Source {
    server::SharedPV pv;
    // handler is borrowed — caller (Cython SharedPV) maintains lifetime
    PyObject* handler;

    GetInterceptSource(server::SharedPV& pv_, PyObject* handler_)
        : pv(pv_), handler(handler_)
    {}

    virtual ~GetInterceptSource() {}

    virtual void onSearch(server::Source::Search& op) override {
        for(auto& name : op) {
            // claim PV only when it is open
            PyLock L;
            if(pv.isOpen())
                name.claim();
        }
    }

    virtual void onCreate(std::unique_ptr<server::ChannelControl>&& op) override {
        // Convert to shared_ptr so we can capture in lambdas
        std::shared_ptr<server::ChannelControl> ctrl(std::move(op));

        // --- onOp: handles GET and PUT ---
        ctrl->onOp([this, ctrl](std::unique_ptr<server::ConnectOp>&& connectop) {
            auto cop = std::shared_ptr<server::ConnectOp>(std::move(connectop));

            // Tell client what data type this PV provides
            {
                PyLock L;
                if(!pv.isOpen()) {
                    cop->error("PV not open");
                    return;
                }
                PyUnlock U;
                cop->connect(pv.fetch());
            }

            // GET handler
            cop->onGet([this](std::unique_ptr<server::ExecOp>&& gop) {
                PyLock L;
                std::shared_ptr<server::ExecOp> op(std::move(gop));
                PyRef pyop(ServerOperation_wrap(op, data::Value{}));

                auto ret(PyRef::allownull(PyObject_CallMethod(handler, "onGet", "O", pyop.obj)));
                if(PyErr_Occurred()) {
                    PySys_WriteStderr("Unhandled Exception %s:%d\n", __FILE__, __LINE__);
                    PyErr_Print();
                    PyErr_Clear();
                    op->error("Internal Error on Remote end");
                }
            });

            // PUT handler — delegate to same pattern as attachHandler
            cop->onPut([this](std::unique_ptr<server::ExecOp>&& rawop, data::Value&& val) {
                PyLock L;
                std::shared_ptr<server::ExecOp> op(std::move(rawop));
                PyRef pyop(ServerOperation_wrap(op, val));

                auto ret(PyRef::allownull(PyObject_CallMethod(handler, "put", "O", pyop.obj)));
                if(PyErr_Occurred()) {
                    PySys_WriteStderr("Unhandled Exception %s:%d\n", __FILE__, __LINE__);
                    PyErr_Print();
                    PyErr_Clear();
                    op->error("Internal Error on Remote end");
                }
            });
        });

        // --- RPC handler ---
        ctrl->onRPC([this](std::unique_ptr<server::ExecOp>&& rawop, data::Value&& val) {
            PyLock L;
            std::shared_ptr<server::ExecOp> op(std::move(rawop));
            PyRef pyop(ServerOperation_wrap(op, val));

            auto ret(PyRef::allownull(PyObject_CallMethod(handler, "rpc", "O", pyop.obj)));
            if(PyErr_Occurred()) {
                PySys_WriteStderr("Unhandled Exception %s:%d\n", __FILE__, __LINE__);
                PyErr_Print();
                PyErr_Clear();
                op->error("Internal Error on Remote end");
            }
        });

        // --- Subscribe handler: deliver current value and track updates ---
        ctrl->onSubscribe([this](std::unique_ptr<server::MonitorSetupOp>&& mop) {
            PyLock L;
            if(!pv.isOpen()) {
                mop->error("PV not open");
                return;
            }
            data::Value proto;
            {
                PyUnlock U;
                proto = pv.fetch();
            }
            auto sub = mop->connect(proto);
            sub->post(proto);
            // Note: future pv.post() calls will NOT automatically propagate to this subscriber.
            // Full subscription support requires Phase 1+ work on the subscription propagation path.
            // For Phase 1, onGet interception is the deliverable; monitors are a best-effort initial connect.
        });
    }
};

std::shared_ptr<server::Source> attachGetHandler(server::SharedPV& pv, PyObject* handler)
{
    return std::make_shared<GetInterceptSource>(pv, handler);
}
```
